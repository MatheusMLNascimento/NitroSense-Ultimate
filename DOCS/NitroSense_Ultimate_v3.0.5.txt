class SignalHub(QObject):
    # Domain-specific signals
    systemHealthChanged = pyqtSignal(str)
    powerModeChanged = pyqtSignal(str)
    thermalStateChanged = pyqtSignal()
    fanProfileChanged = pyqtSignal(str)
    hardwareErrorDetected = pyqtSignal(ErrorCode)
    emergencyProtocolTriggered = pyqtSignal(ErrorCode)
    watchdogTriggered = pyqtSignal()
    
    def subscribe(self, signal_name: str, callback: Callable):
        """Register callback for named signal"""
        getattr(self, signal_name).connect(callback)
    
    def emit_event(self, signal_name: str, *args):
        """Emit event via signal hub"""
        getattr(self, signal_name).emit(*args)

# Usage Example:
hub = get_signal_hub()
hub.subscribe("systemHealthChanged", my_callback)

class ThreadSafeStateMachine(QObject):
    _lock = QReadWriteLock()
    _state = {
        "cpu_temp": 0.0,
        "gpu_temp": 0.0,
        "fan_rpm": 0,
        "fan_speed_percent": 0,
        "power_mode": "balanced",
        "thermal_state": "normal",
        "emergency_active": False,
        "watchdog_fired": False
    }
    
    def read_state(self, key: str) -> Any:
        """Atomic read with read lock"""
        self._lock.lockForRead()
        try:
            return copy.copy(self._state.get(key))
        finally:
            self._lock.unlock()
    
    def update_state(self, key: str, value: Any):
        """Atomic write with write lock"""
        self._lock.lockForWrite()
        try:
            self._state[key] = value
        finally:
            self._lock.unlock()

# Usage Example:
state = get_state_machine()
cpu_temp = state.read_state("cpu_temp")  # Atomic read
state.update_state("fan_speed_percent", 50)  # Atomic write

class DirtyBitCache:
    def __init__(self):
        self._cache = {}
        self._dirty = set()
    
    def update_cache(self, key: str, new_value: float) -> bool:
        """Returns True if value changed (accounting for tolerance)"""
        old_value = self._cache.get(key)
        
        # 0.1°C tolerance for temp values (prevent flicker)
        if isinstance(new_value, float) and isinstance(old_value, float):
            if abs(new_value - old_value) < 0.1:
                return False
        
        if old_value != new_value:
            self._cache[key] = new_value
            self._dirty.add(key)
            return True
        return False
    
    def is_dirty(self, key: str) -> bool:
        """Returns True if changed, then resets"""
        result = key in self._dirty
        if result:
            self._dirty.discard(key)
        return result

# Usage Example (UI Page):
cache = get_dirty_bit_cache()

# In monitoring loop:
cache.update_cache("cpu_temp", 45.5)

# In UI update:
if cache.is_dirty("cpu_temp"):
    self.lcd_cpu.setText(f"{cpu_temp:.1f}°C")
    self.lcd_cpu.repaint()  # Only repaints if changed!

class SystemIntegrityCheck:
    @staticmethod
    def full_integrity_check() -> Dict[str, Any]:
        """Run all 3 levels of diagnostics"""
        
        # LEVEL 1: Binary availability
        binaries = {
            "nbfc": shutil.which("nbfc") is not None,
            "nvidia-smi": shutil.which("nvidia-smi") is not None,
            "sensors": shutil.which("sensors") is not None,
        }
        
        # LEVEL 2: Kernel parameters
        ec_writable = False
        try:
            with open("/sys/module/ec_sys/parameters/write_support", "r") as f:
                value = f.read().strip()
                ec_writable = (value == "Y")
            
            # Auto-enable if not writable
            if not ec_writable:
                subprocess.run(["sudo", "modprobe", "-r", "ec_sys"], check=False)
                subprocess.run(["sudo", "modprobe", "ec_sys", "write_support=1"], 
                             check=False)
        except Exception as e:
            logger.warning(f"EC check: {e}")
        
        # LEVEL 3: Python modules
        modules = {
            "psutil": _check_module("psutil"),  # Critical
            "PyQt6": _check_module("PyQt6"),     # Critical
            "matplotlib": _check_module("matplotlib"),  # Optional (lazy)
            "numpy": _check_module("numpy"),            # Optional (lazy)
        }
        
        return {
            "status": "OK" if all([binaries, ec_writable]) else "WARNING",
            "binaries": binaries,
            "ec_writable": ec_writable,
            "modules": modules
        }

# Usage Example (main.py):
integrity = SystemIntegrityCheck.full_integrity_check()
if "CRITICAL" not in integrity["status"]:
    logger.error("System not ready!")
else:
    logger.info("✅ System ready")

class HardwareWatchdog(QThread):
    timeout_detected = pyqtSignal()  # Emitted on 10s timeout
    
    def __init__(self, timeout_sec: int = 10):
        super().__init__()
        self.timeout_sec = timeout_sec
        self.last_heartbeat = time.time()
        self.running = True
    
    def heartbeat(self):
        """Call from monitoring loop to signal 'I'm alive'"""
        self.last_heartbeat = time.time()
    
    def run(self):
        """Independent thread - checks every 1 second"""
        while self.running:
            time.sleep(1)
            elapsed = time.time() - self.last_heartbeat
            
            if elapsed > self.timeout_sec:
                logger.critical(f"🚨 Watchdog timeout ({elapsed:.1f}s elapsed)")
                self.timeout_detected.emit()
                self._emergency_bus_reset()
                self.last_heartbeat = time.time()  # Reset
    
    def _emergency_bus_reset(self):
        """Recover EC bus"""
        try:
            # 1. Kill NBFC
            os.system("sudo pkill -9 nbfc")
            time.sleep(1)
            
            # 2. Reload EC module
            os.system("sudo modprobe -r ec_sys")
            os.system("sudo modprobe ec_sys write_support=1")
            time.sleep(2)
            
            # 3. Restart NBFC
            os.system("sudo systemctl restart nbfc_service")
            
            logger.info("✅ Bus recovery complete")
        except Exception as e:
            logger.error(f"Bus reset failed: {e}")
    
    def stop(self):
        """Graceful shutdown"""
        self.running = False
        self.wait()

# Usage Example (system.py):
watchdog = HardwareWatchdog(timeout_sec=10)
watchdog.timeout_detected.connect(on_emergency_protocol)
watchdog.start()

# In monitoring loop:
watchdog.heartbeat()  # Call after each sensor read
